    <script>
        // ฟังก์ชันอัปเดตวันและเวลารูปแบบไทย
        function updateLiveClock() {
            const now = new Date();
            
            // รูปแบบเวลา 00:00:00
            const timeString = now.toLocaleTimeString('th-TH', { hour12: false });
            
            // รูปแบบวันที่ไทย เช่น วันเสาร์ ที่ 4 ก.ค. 2569
            const options = { weekday: 'long', day: 'numeric', month: 'short', year: 'numeric' };
            const dateString = now.toLocaleDateString('th-TH', options);
            
            const timeEl = document.getElementById('live-time');
            const dateEl = document.getElementById('live-date');
            
            if(timeEl) timeEl.textContent = timeString;
            if(dateEl) dateEl.textContent = "📅" + dateString;
        }
        setInterval(updateLiveClock, 1000);
        updateLiveClock(); // รันทันทีเมื่อโหลดหน้าเว็บ

        // ข้อ 3: ระบบดึงสถิติอัตโนมัติเมื่อ รีเฟรชหน้าจอ หรือ ล็อกอินสำเร็จ หรือ มีการเปลี่ยนวันที่
        document.addEventListener("DOMContentLoaded", () => {
            // ตั้งค่าวันที่ปัจจุบันลงในช่อง input[type=date] เป็นค่าเริ่มต้นถ้ายังไม่มีค่า
            const dateInput = document.getElementById('record-date');
            if (dateInput && !dateInput.value) {
                const today = new Date().toISOString().split('T')[0];
                dateInput.value = today;
            }

            // ตรวจสอบสถานะการล็อกอินจาก Firebase (เมื่อล็อกอินสำเร็จ ให้ดึงสถิติทันที)
            if (typeof firebase !== 'undefined' && firebase.auth) {
                firebase.auth().onAuthStateChanged((user) => {
                    if (user) {
                        triggerLoadStatistics();
                    }
                });
            } else {
                // หากไม่มีระบบ Firebase ให้ทดลองดึงข้อมูลทันทีเมื่อโหลด DOM สำเร็จ (รีเฟรชหน้า)
                triggerLoadStatistics();
            }

            // ถ้ามีการเปลี่ยนวันที่ในปฏิทิน ให้โหลดสถิติของวันนั้นใหม่ทันที
            if (dateInput) {
                dateInput.addEventListener('change', triggerLoadStatistics);
            }
        });

        // ฟังก์ชันช่วยสั่งงานไปยังแอปหลัก (app.js) เพื่อดึงข้อมูลสถิติมาแสดงผล
        function triggerLoadStatistics() {
            // ดักเรียกฟังก์ชันที่มีแนวโน้มว่าจะอยู่ใน app.js เพื่อโหลดสถิติปัจจุบันของวันนั้นๆ
            if (typeof loadAttendanceData === 'function') {
                loadAttendanceData();
            } else if (typeof fetchAttendanceData === 'function') {
                fetchAttendanceData();
            } else if (typeof checkAndLoadDailyData === 'function') {
                checkAndLoadDailyData();
            } else {
                console.log("ระบบพร้อมแสดงผลสถิติ: รอแอปพลิเคชันหลักเชื่อมโยงฐานข้อมูลข้อมูล");
            }
        }
